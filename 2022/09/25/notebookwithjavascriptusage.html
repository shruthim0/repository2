function Fruit(name , color) {
    this.name = name;
    this.color = color;
    this.role = "";
}

Fruit.prototype.setRole = function(role) {
    this.role = role;
}

Fruit.prototype.toJSON = function() {
    const obj = {name: this.name, color: this.color};
    const json = JSON.stringify(obj);
    return json;
}
